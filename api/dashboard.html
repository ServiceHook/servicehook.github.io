üöÄ Quick Integration

Send a POST request to create a short link.

JavaScript (Fetch) cURL 
const createLink = async () => {
  const res = await fetch('https://jachu.xyz/api/create', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-API-Key': 'YOUR_API_KEY'
    },
    body: JSON.stringify({ 
      url: 'https://example.com/my-long-link',
      slug: 'custom-alias' // Optional
    })
  });

  const data = await res.json();
  console.log(data.short_url);
};
curl -X POST https://jachu.xyz/api/create \
  -H "Content-Type: application/json" \
  -H "X-API-Key: YOUR_API_KEY" \

    "slug": "my-google"
