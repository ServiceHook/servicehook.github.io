const shorten = async (longUrl) => {
  const res = await fetch('https://jachu.xyz/api/create', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-API-Key': 'YOUR_API_KEY'
    },
    body: JSON.stringify({ 
      url: longUrl,
      slug: 'my-alias' // Optional custom alias
    })
  });

  const data = await res.json();

  }
};
